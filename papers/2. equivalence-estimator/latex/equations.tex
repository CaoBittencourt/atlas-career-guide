% linear-logistic trigonometrically-scaled equivalence estimator
\newcommand{\EquivalenceEstimator}{
\begin{gather}
    \label{eqn:equivalence_estimator}
    \eq(x,M) = 
        x\left\{
            1 + M (1 - x) \exp[-
                b(x - M)
            ]
        \right\} ^ {
            -\frac{M}{x}
    },\\
    b = 
    \tan\bigg[
        \frac{\pi}{2}
        \cos^{M(1 - M)}
        % \cos^{x(1 - M)}
            \bigg(
                \frac{\pi}{2} 
                % M (1 - M)
                x (1 - M)
            \bigg)
    \bigg],\\
    x, M \in [0,1].
\end{gather}
