% Professional Attribute
\newcommand{\EqnAttribute}{
    \begin{gather}
    \label{eqn:attribute}
        a_{i}^{\upsilon} \in [0,100] \ \forall \ i, \upsilon
    ,\end{gather}
}

% Skill Set
\newcommand{\EqnSkillSet}{
    \begin{gather}
    \label{eqn:skill_set}
        \vb{a_{\boldsymbol{\upsilon}}} = (a_{1}^{\upsilon}, ..., a_{m}^{\upsilon})
    .\end{gather}
}

% Euclidean distance
\newcommand{\EqnEuclideanDistance}{
    \begin{gather}
        \label{eqn:euclidean_distance}
            d(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}) = 
            \norm{\vb{a_{\boldsymbol{\upsilon}}} - \vb{a_{q}}} = 
            \sqrt{
            \sum_{i = 1}^{m}{
            (a_{i}^{\upsilon} - a_{i}^{q})^2
            }
            }
        .\end{gather}
}

% Normalized Euclidean distance
\newcommand{\EqnNormalizedDistance}{
    \begin{gather}
    \label{eqn:normalized_distance}
        \tilde{d}(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}) = 
        \sqrt{
        \frac{
        \sum_{i = 1}^{m}{
        (a_{i}^{\upsilon} - a_{i}^{q})^2
        }
        }{
        \sum_{i = 1}^{m}{
        \max(100 - a_{i}^{q}, a_{i}^{q})^2
        }
        }
        }
    .\end{gather}
}

% Similarity
\newcommand{\EqnSimilarity}{
    \begin{gather}
    \label{eqn:similarity}
        s(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}) = 1 - \tilde{d}(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}) 
    .\end{gather}
}

% Weighted Euclidean distance
\newcommand{\EqnWeightedDistance}{
    \begin{gather}
    \label{eqn:weighted_distance}
        \tilde{d}_w(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}) = 
        \sqrt{
        \frac{
        \sum_{i = 1}^{m}{
        a_{i}^{q} (a_{i}^{\upsilon} - a_{i}^{q})^2
        }
        }{
        \sum_{i = 1}^{m}{
        a_{i}^{q} \max(100 - a_{i}^{q}, a_{i}^{q})^2
        }
        }
        }
    .\end{gather}
}

\newcommand{\EqnWeightedSimilarity}{
    \begin{gather}
    \label{eqn:weighted_similarity}
        s_w(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}) = 1 - \tilde{d}_w(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}})
    ,\end{gather}
}

% Distance derivative
\newcommand{\EqnDistanceDerivative}{
    \begin{gather}
    \label{eqn:distance_derivative}
        a_{i}^{\upsilon} > a_{i}^{q} \iff
        \frac{
        \partial{\tilde{d}(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}})}}
        {\partial{a_{i}^{\upsilon}}} > 0 \iff
        \frac{
        \partial{s(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}})}}
        {\partial{a_{i}^{\upsilon}}} < 0
    .\end{gather}
}

% Unweighted vs Weighted distance
\newcommand{\EqnUnweightedWeightedDistances}{
    \begin{gather}
    \label{eqn:unweighted_vs_weighted_distance}
        \tilde{d}_w(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}) \leq
        \tilde{d}(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}) \ \forall \ \vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}
    .\end{gather}
}

% Similarity symmetry
\newcommand{\SimilarityAsymmetry}{
    \begin{gather}
    \label{eqn:similarity_asymmetry}
        s(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}) = s(\vb{a_{q}}, \vb{a_{\boldsymbol{\upsilon}}}) \iff \vb{a_{\boldsymbol{\upsilon}}} = \vb{a_{q}}.
    \end{gather}
}

% Similarity scaling
\newcommand{\SimilarityScaling}{
    \begin{gather}
    \label{eqn:similarity_scaling}
        s_w(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}}) = 1 - f(\tilde{d}_w(\vb{a_{\boldsymbol{\upsilon}}}, \vb{a_{q}})),
    \end{gather}
}
